Description

A responsive button that follows Material 3 design guidelines for primary actions.

Use PrimaryButton for the most important action on a screen, such as "Submit", "Confirm", or "Create". It features a filled background with the primary color, providing maximum emphasis and discoverability.

The button automatically adapts its padding, font size, and touch targets based on the ButtonSize parameter. It supports all standard Material interactions including ripple effects, disabled states, and loading states.

/// A responsive button that follows Material 3 guidelines.
///
/// Use [PrimaryButton] for main actions, [SecondaryButton] for less important ones.
///
/// The button automatically adapts padding and font size based on [ButtonSize].
///
/// See also:
///  * [SecondaryButton], a lower-emphasis button.
///  * [ButtonSize], for controlling button sizing.
class PrimaryButton extends StatelessWidget {
  const PrimaryButton({
    Key? key,
    required this.onPressed,
    required this.child,
    this.size = ButtonSize.medium,
    this.isLoading = false,
    this.fullWidth = false,
  }) : super(key: key);

  final VoidCallback? onPressed;
  final Widget child;
  final ButtonSize size;
  final bool isLoading;
  final bool fullWidth;

  @override
  Widget build(BuildContext context) {
    final theme = Theme.of(context);
    final color = onPressed != null 
      ? theme.primaryColor 
      : Colors.grey[400];

    return SizedBox(
      width: fullWidth ? double.infinity : null,
      child: MaterialButton(
        onPressed: isLoading ? null : onPressed,
        padding: _getPadding(),
        shape: RoundedRectangleBorder(
          borderRadius: BorderRadius.circular(8),
        ),
        color: color,
        child: isLoading
          ? SizedBox(
              width: 20,
              height: 20,
              child: CircularProgressIndicator(
                strokeWidth: 2,
                valueColor: AlwaysStoppedAnimation(Colors.white),
              ),
            )
          : child,
      ),
    );
  }

  EdgeInsets _getPadding() {
    switch (size) {
      case ButtonSize.small:
        return EdgeInsets.symmetric(horizontal: 12, vertical: 8);
      case ButtonSize.medium:
        return EdgeInsets.symmetric(horizontal: 16, vertical: 12);
      case ButtonSize.large:
        return EdgeInsets.symmetric(horizontal: 24, vertical: 16);
    }
  }
}

enum ButtonSize { small, medium, large }
          
        
// Generated JavaScript equivalent
class PrimaryButton {
  constructor(props = {}) {
    this.onPressed = props.onPressed;
    this.child = props.child;
    this.size = props.size || 'medium';
    this.isLoading = props.isLoading || false;
    this.fullWidth = props.fullWidth || false;
  }

  getPadding() {
    const sizes = {
      small: { horizontal: 12, vertical: 8 },
      medium: { horizontal: 16, vertical: 12 },
      large: { horizontal: 24, vertical: 16 },
    };
    return sizes[this.size] || sizes.medium;
  }

  render() {
    const padding = this.getPadding();
    const isDisabled = !this.onPressed || this.isLoading;
    
    const button = document.createElement('button');
    button.className = `primary-button primary-button--${this.size}`;
    button.style.padding = `${padding.vertical}px ${padding.horizontal}px`;
    button.disabled = isDisabled;
    
    if (this.fullWidth) {
      button.style.width = '100%';
    }

    button.onclick = (e) => {
      if (!isDisabled && this.onPressed) {
        this.onPressed(e);
      }
    };

    if (this.isLoading) {
      const spinner = document.createElement('div');
      spinner.className = 'spinner';
      button.appendChild(spinner);
    } else {
      button.appendChild(this.child.render());
    }

    return button;
  }
}

export { PrimaryButton, ButtonSize };
